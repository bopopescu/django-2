def home(request):
	#select * from lesson
    all_Class = Lesson.objects.all()
    #select * from Lesson, Lessonlevel  where  Lessonlevel.level_name = 'Beginner'
    begin_Level = Lessonlevel.objects.get(level_name='Beginner').lesson_set.all()
    #select * from Lesson, Lessonlevel where lesson.class_name='Python 101'
    level_of_class = Lessonlevel.objects.get(lesson__class_name='Python 101')
    context = {'all_Class' : all_Class,
    			'beginner_Level': begin_Level,
    			'level_of_class': level_of_class
               }

    return render(request, 'oneplace/index.html', context)
